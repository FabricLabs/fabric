                for (let j=0; j<16; j++) { // encode 4 chars per integer (64 per block), big-endian encoding
                    M[i][j] = (msg.charCodeAt(i*64+j*4+0)<<24) | (msg.charCodeAt(i*64+j*4+1)<<16)

            M[N-1][14] = Math.floor(lenHi);

            function utf8Encode(str) {
                try {
